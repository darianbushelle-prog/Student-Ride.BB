3 Database Schema

PostgreSQL 16 hosted on Supabase. All tables use UUID primary keys, created_at / updated_at timestamps, and Row Level Security policies. PostGIS is enabled for geospatial stop coordinates.

-- Students can only see their own data
CREATE POLICY "students_own_data" ON students
  FOR SELECT USING (user_id = auth.uid());

-- Students can only read their own notification receipts
CREATE POLICY "own_receipts" ON notification_receipts
  FOR ALL USING (user_id = auth.uid());

-- Admins bypass RLS (via service role key in API layer)
CREATE POLICY "admin_all" ON students
  FOR ALL USING (
    EXISTS (SELECT 1 FROM users
             WHERE id = auth.uid() AND role = 'admin')
  );

4 REST API Design

All endpoints are prefixed with /api/v1. Authentication uses Bearer JWT tokens issued by Supabase Auth. Admin routes require role: 'admin' in the JWT claims.

⚠️
Rate limiting Apply express-rate-limit: 100 req/min per IP for public endpoints, 1000 req/min for authenticated student sessions, and 10 req/min on the login endpoint to prevent brute force.

5 SMS Integration — Digicel Barbados

Digicel is the dominant carrier in Barbados (~65% market share). Their bulk SMS API supports two-way messaging, sender ID registration, and delivery webhooks. FLOW (C&W) students are also reachable via the same gateway.

⚠️
Registration required You must register your sender ID ("SchoolRide") with Digicel before going live. This takes 5–10 business days. Contact: enterprise@digicelgroup.com with your MOE authorisation letter.

SMS Service

JavaScriptsrc/services/sms.service.js
const axios = require('axios');
const { createHmac } = require('crypto');
const { supabase } = require('../lib/supabase');

const DIGICEL_URL    = process.env.DIGICEL_API_URL;
const DIGICEL_KEY    = process.env.DIGICEL_API_KEY;
const SENDER_ID      = process.env.DIGICEL_SENDER_ID; // 'SchoolRide'
const MAX_BATCH_SIZE = 500; // Digicel bulk limit per request

/**
 * Send an alert to all students on a route (or all routes).
 * Batches in groups of 500 to respect gateway limits.
 */
async function dispatchSMSAlert(notificationId) {
  // 1. Fetch notification + recipient list from DB
  const { data: notif } = await supabase
    .from('notifications')
    .select('*, routes(id, name)')
    .eq('id', notificationId)
    .single();

  let query = supabase
    .from('students')
    .select('id, users(phone, sms_opt_in)')
    .eq('is_active', true)
    .not('users.phone', 'is', null)
    .eq('users.sms_opt_in', true);

  // Filter by route if not a broadcast
  if (notif.route_id) query = query.eq('route_id', notif.route_id);

  const { data: students } = await query;
  const recipients = students.map(s => s.users.phone);

  // 2. Create per-student receipt rows (status: 'pending')
  await supabase.from('notification_receipts').insert(
    students.map(s => ({
      notification_id: notificationId,
      user_id: s.id,
      channel: 'sms',
      status: 'pending'
    }))
  );

  // 3. Batch and send via Digicel
  const batches = chunkArray(recipients, MAX_BATCH_SIZE);
  const results = [];

  for (const batch of batches) {
    const payload = {
      sender: SENDER_ID,
      recipients: batch,
      message: notif.message,
      callback_url: `${process.env.API_BASE_URL}/api/v1/webhooks/sms-delivery`,
      reference: notificationId
    };

    const response = await axios.post(
      `${DIGICEL_URL}/messages/bulk`,
      payload,
      { headers: { 'X-API-Key': DIGICEL_KEY, 'Content-Type': 'application/json' } }
    );
    results.push(response.data);
  }

  // 4. Update notification with delivery stats
  await supabase.from('notifications').update({
    sent_at: new Date().toISOString(),
    recipient_count: recipients.length
  }).eq('id', notificationId);

  return { sent: recipients.length, batches: batches.length };
}

/**
 * Validate Digicel webhook signature and update delivery receipt.
 * Call this from POST /api/v1/webhooks/sms-delivery
 */
async function handleDeliveryReceipt(req, res) {
  // Verify HMAC signature from Digicel header
  const sig       = req.headers['x-digicel-signature'];
  const expected  = createHmac('sha256', DIGICEL_KEY)
                     .update(JSON.stringify(req.body))
                     .digest('hex');
  if (sig !== expected) return res.sendStatus(401);

  const { message_id, status, recipient } = req.body;
  const dbStatus = statusMap[status] ?? 'failed';

  await supabase.from('notification_receipts')
    .update({ status: dbStatus, gateway_message_id: message_id })
    .eq('gateway_message_id', message_id);

  res.sendStatus(200);
}

const statusMap = {
  DELIVRD: 'delivered',
  UNDELIV: 'failed',
  EXPIRED: 'failed',
  REJECTED: 'failed'
};

const chunkArray = (arr, size) =>
  Array.from({ length: Math.ceil(arr.length / size) }, (_, i) =>
    arr.slice(i * size, i * size + size));

module.exports = { dispatchSMSAlert, handleDeliveryReceipt };

// Alert type → notification colour (Android accent)
const TYPE_COLORS = {
  delay:   '#D97706', cancel: '#E8523A',
  driver:  '#0B7C6E', stop:   '#003F87',
  holiday: '#7C3AED', general:'#4A5568'
};

/**
 * Send FCM push notifications to all eligible students.
 * Uses sendEachForMulticast for up to 500 tokens per call.
 */
async function dispatchPushAlert(notificationId) {
  const { data: notif } = await supabase
    .from('notifications')
    .select('*')
    .eq('id', notificationId)
    .single();

  // Fetch FCM tokens for recipients
  let query = supabase
    .from('students')
    .select('id, users(id, fcm_token, fcm_token_updated_at)')
    .eq('is_active', true)
    .not('users.fcm_token', 'is', null);

  if (notif.route_id) query = query.eq('route_id', notif.route_id);

  const { data: students } = await query;

  // Filter out stale tokens (> 60 days old)
  const cutoff = new Date(Date.now() - 60 * 24 * 60 * 60 * 1000);
  const eligible = students.filter(s =>
    s.users.fcm_token_updated_at > cutoff.toISOString()
  );

  const tokens = eligible.map(s => s.users.fcm_token);
  if (!tokens.length) return { sent: 0 };

  // Build FCM multicast message
  const message = {
    notification: {
      title: notif.title,
      body:  notif.message.substring(0, 200)  // FCM body limit
    },
    data: {
      notification_id: notificationId,
      alert_type:      notif.alert_type,
      route_id:        notif.route_id ?? '',
      click_action:    'OPEN_NOTIFICATIONS'   // handled in the app
    },
    android: {
      priority: notif.urgency === 'high' ? 'high' : 'normal',
      notification: {
        color: TYPE_COLORS[notif.alert_type] ?? '#003F87',
        channel_id: 'schoolride_alerts',
        sound: 'default'
      }
    },
    apns: {  // iOS
      payload: { aps: { sound: 'default', badge: 1 } }
    }
  };

  // Send in batches of 500 (FCM multicast limit)
  const batches = chunkArray(tokens, 500);
  let successCount = 0, failureCount = 0;
  const stalTokens = [];

  for (const batch of batches) {
    const res = await admin.messaging().sendEachForMulticast({
      tokens: batch, ...message
    });
    successCount += res.successCount;
    failureCount += res.failureCount;

    // Collect invalid/expired tokens for cleanup
    res.responses.forEach((r, i) => {
      if (!r.success && ['messaging/invalid-registration-token',
                         'messaging/registration-token-not-registered']
                        .includes(r.error?.code)) {
        staleTokens.push(batch[i]);
      }
    });
  }

  // Clean up stale tokens in DB
  if (staleTokens.length) {
    await supabase.from('users')
      .update({ fcm_token: null })
      .in('fcm_token', staleTokens);
  }

  return { sent: tokens.length, successCount, failureCount };
}

const chunkArray = (arr, size) =>
  Array.from({ length: Math.ceil(arr.length / size) }, (_, i) =>
    arr.slice(i * size, i * size + size));

module.exports = { dispatchPushAlert };

export async function registerPushNotifications(accessToken) {

  const permission = await Notification.requestPermission();
  if (permission !== 'granted') return;

  const token = await getToken(messaging, {
    vapidKey: process.env.NEXT_PUBLIC_FIREBASE_VAPID_KEY

  onMessage(messaging, payload => {
    showInAppNotification(payload.notification);
  });
}

7 Authentication & SSO

Supabase Auth handles JWT generation and refresh. For the Ministry of Education SSO path (single login for all government education systems), SAML 2.0 is the standard.

1
Student self-registration or admin bulk import
Admin uploads a CSV of students → API creates auth.users entries via Supabase Admin SDK, then inserts rows into users and students tables. Students receive a welcome SMS with their temporary password.
2
JWT issued with custom claims
On login, Supabase issues a signed JWT. A database trigger populates a raw_app_meta_data claim with { "role": "student", "student_id": "uuid" } — available in all API calls via auth.jwt().
3
Row Level Security enforces data isolation
Every Supabase query from the student app runs with their JWT. RLS policies ensure they can only read their own rows — no extra application-layer checks needed for basic data access.
4
Ministry SSO via SAML 2.0 (Phase 2)
Supabase supports SAML 2.0 as an IdP. The Ministry's existing Active Directory / Azure AD becomes the identity provider. Students log in with their school Microsoft 365 credentials — no separate password needed.
🔐
Refresh token strategy Access tokens expire in 1 hour. Refresh tokens last 7 days. On the client, intercept 401 responses, call /auth/v1/token?grant_type=refresh_token, and retry the original request. Supabase's JS client handles this automatically.

// Triggered by Supabase DB webhook on notifications INSERT
serve(async (req) => {
  const { record } = await req.json(); // The new notification row
  const supabase = createClient(
    Deno.env.get("SUPABASE_URL")!,
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
  );

  // Skip if scheduled for future
  if (record.scheduled_for && new Date(record.scheduled_for) > new Date()) {
    return new Response("Scheduled — skipping", { status: 200 });
  }

  // Fan out to SMS and/or Push workers in parallel
  const dispatches = [];
  if (record.channels.includes('sms')) {
    dispatches.push(
      fetch(`${Deno.env.get("API_BASE_URL")}/api/v1/internal/sms-dispatch`, {
        method: 'POST',
        body: JSON.stringify({ notification_id: record.id }),
        headers: { 'X-Internal-Key': Deno.env.get('INTERNAL_API_KEY')! }
      })
    );
  }
  if (record.channels.includes('push')) {
    dispatches.push(
      fetch(`${Deno.env.get("API_BASE_URL")}/api/v1/internal/push-dispatch`, {
        method: 'POST',
        body: JSON.stringify({ notification_id: record.id }),
        headers: { 'X-Internal-Key': Deno.env.get('INTERNAL_API_KEY')! }
      })
    );
  }

  await Promise.allSettled(dispatches);
  return new Response("Dispatched", { status: 200 });
});

9 Deployment Architecture

A minimal, cost-effective production setup for a government digital service in Barbados. Estimated monthly cost at full scale: BBD $120–200/month.

yamlInfrastructure overview
# ─── DNS ────────────────────────────────────────
schoolridebb.gov.bb     → Vercel (Next.js frontend)
api.schoolridebb.gov.bb → Railway (Node.js API, auto-deploys on git push)
db.schoolridebb.gov.bb  → Supabase (managed PostgreSQL, backups daily)

# ─── CI / CD (GitHub Actions) ───────────────────
on push to main:
  - run tests
  - deploy frontend  → vercel --prod
  - deploy API       → railway up
  - deploy functions → supabase functions deploy

# ─── Monitoring ─────────────────────────────────
uptime:   UptimeRobot (free, 5-min checks → SMS alert to Transport Board)
errors:   Sentry free tier (JS + Node error tracking)
logs:     Railway built-in log viewer + Supabase log dashboard
metrics:  Supabase dashboard (query performance, DB size, auth events)
